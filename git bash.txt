git bash

pwd - текущая директория
cd ~ - переход в домашнюю
cd .. - подняться в папку выше
cd ./<имя директории>/ - переход в опр директорию ниже
cd /c/users/vika/ - переход по опр пути

гит баш умненький и подсказывает по табу
ls - список папок и файлов
touch <имя файла> - создание файла
mkdir <имя дир> - создание директории
rm <имя файла> - удаление файла
rmdir <имя дир> - удаление дир
cat - посмотреть содержимое файла
cp <имя файла> <новое имя файла> - копия файла
mv <имя файла> - перемещение
if хотим удалить репозиторий, просто удаляем .git
git status - статус репозитория
git commit - закоммитить
git log - просмотр предыдущих коммитов

изменённый - не подг к сохр
индексированный - подг но не сохр
закомиченный - сохр
